// ── AI HELPERS ────────────────────────────────────────────────────────────────
async function callAI(system, content) {
  const r = await fetch("https://api.anthropic.com/v1/messages", {
    method:"POST", headers:{"Content-Type":"application/json"},
    body:JSON.stringify({ model:"claude-sonnet-4-20250514", max_tokens:2000, system, messages:[{role:"user",content}] })
  });
  return (await r.json()).content?.[0]?.text || "";
}
async function callJSON(system, content) {
  try {
    const text = await callAI(system, content);
    // Strip markdown code fences if present
    const clean = text.replace(/```json|```/g, "").trim();
    return JSON.parse(clean);
  } catch { return null; }
}

// ── DAILY QUIZ MODAL ──────────────────────────────────────────────────────────
// AI-generated questions from vault content + due spaced repetition cards
const DailyQuizModal = ({ vaults, setVaults, onClose }) => {
  const T = useT();
  const [phase, setPhase] = useState("loading");
  const [deck, setDeck] = useState([]);
  const [idx, setIdx] = useState(0);
  const [shown, setShown] = useState(false);
  const [log, setLog] = useState([]);
  const [anim, setAnim] = useState(null);
  const [aiQuestions, setAiQuestions] = useState([]);

  useEffect(() => { buildDeck(); }, []);

  const buildDeck = async () => {
    const allCards = [];
    Object.values(vaults).forEach(v => {
      const sub = getSub(v.subject);
      [...v.flashcards, ...v.words.map(w=>({...w, front:w.word, back:w.definition, type:"word"}))].forEach(c => {
        allCards.push({ ...c, vaultId:v.id, vaultName:v.name, sub, type:c.type||"card" });
      });
    });

    // Due cards first (spaced repetition)
    const due = allCards.filter(c => isDue(c)).sort(() => Math.random()-0.5).slice(0,5);
    // Weak cards (low mastery) next
    const weak = allCards.filter(c => !isDue(c) && (c.mastered||0) < 2).sort((a,b)=>(a.mastered||0)-(b.mastered||0)).slice(0,3);

    let combined = [...due, ...weak];

    // If not enough cards, generate AI questions from vault content
    if (allCards.length > 0 && combined.length < 5) {
      const sample = allCards.slice(0,8).map(c=>`Q: ${c.front} A: ${c.back}`).join("\n");
      const sys = `You are a study quiz AI. Generate ${5-combined.length} NEW quiz questions (not duplicates) from this content. Return ONLY valid JSON array:
[{"front":"<question>","back":"<answer>","type":"ai"}]`;
      const aiCards = await callJSON(sys, `Content:\n${sample}`);
      if (Array.isArray(aiCards)) {
        aiCards.forEach(c => combined.push({...c, id:"ai-"+Date.now()+Math.random(), mastered:0, sub:getSub(Object.values(vaults)[0]?.subject||"reading")}));
      }
    }

    if (combined.length === 0) { setPhase("empty"); return; }
    setDeck(combined.slice(0,8));
    setPhase("quiz");
  };

  const answer = (got) => {
    setAnim(got?"right":"left");
    const card = deck[idx];
    const newLog = [...log, {...card, got}];

    // Update mastery + spaced repetition schedule
    if (card.vaultId) {
      setVaults(prev => {
        const v = prev[card.vaultId];
        if (!v) return prev;
        const newMastery = got ? Math.min((card.mastered||0)+1,5) : Math.max((card.mastered||0)-1,0);
        const bump = arr => arr.map(c => c.id!==card.id?c:{...c, mastered:newMastery, nextReview:getNextReview(newMastery)});
        return {...prev,[card.vaultId]:{...v,
          flashcards:card.type==="card"?bump(v.flashcards):v.flashcards,
          words:card.type==="word"?bump(v.words):v.words,
        }};
      });
    }

    setTimeout(()=>{ setAnim(null); if(idx+1>=deck.length){setLog(newLog);setPhase("done");}else{setIdx(n=>n+1);setShown(false);} }, 280);
  };

  const dueCount = Object.values(vaults).reduce((a,v)=>{
    return a+[...v.flashcards,...v.words].filter(c=>isDue(c)).length;
  },0);

  return (
    <div style={{ position:"fixed", inset:0, background:T.bg, zIndex:200, overflowY:"auto" }}>
      <div style={{ display:"flex", alignItems:"center", gap:12, padding:"18px 20px 14px", borderBottom:`1px solid ${T.border}`, background:T.card, position:"sticky", top:0 }}>
        <button onClick={onClose} style={{ background:"none", border:`1px solid ${T.border}`, borderRadius:50, width:34, height:34, cursor:"pointer", color:T.muted, fontSize:16, display:"flex", alignItems:"center", justifyContent:"center" }}>×</button>
        <div>
          <div style={{ fontWeight:800, fontSize:17, color:T.text }}>Daily Quiz</div>
          <div style={{ fontSize:11, color:T.muted }}>{dueCount} cards due · Spaced Repetition active</div>
        </div>
      </div>

      <div style={{ padding:"20px" }}>
        {phase==="loading" && <div style={{ textAlign:"center", padding:"40px 0" }}><div style={{ fontSize:36, marginBottom:12 }}>⏳</div><div style={{ color:T.muted }}>Building your personalised quiz...</div></div>}

        {phase==="empty" && <div style={{ textAlign:"center", padding:"40px 0", color:T.faint }}><div style={{ fontSize:40, marginBottom:12 }}>📭</div><div style={{ fontWeight:700, fontSize:15, color:T.text, marginBottom:6 }}>No cards yet!</div><div style={{ fontSize:13 }}>Create vaults and add flashcards to start your daily quiz.</div></div>}

        {phase==="quiz" && deck.length>0 && (() => {
          const card=deck[idx]; const sub=card.sub||getSub("reading");
          const dueTag = isDue(card) && card.type!=="ai";
          return (
            <div>
              <div style={{ display:"flex", justifyContent:"space-between", alignItems:"center", marginBottom:16 }}>
                <div style={{ fontSize:12, color:T.muted }}>{idx+1} of {deck.length}</div>
                {dueTag && <span style={{ fontSize:10, color:T.accent, background:T.accentBg, border:`1px solid ${T.accent}44`, borderRadius:50, padding:"3px 10px", letterSpacing:1, textTransform:"uppercase" }}>Due for Review</span>}
                {card.type==="ai" && <span style={{ fontSize:10, color:T.blue, background:T.blueBg, border:`1px solid ${T.blueBd}`, borderRadius:50, padding:"3px 10px", letterSpacing:1, textTransform:"uppercase" }}>AI Question</span>}
              </div>
              <div style={{ background:T.card2, borderRadius:50, height:5, marginBottom:20, overflow:"hidden" }}>
                <div style={{ width:`${((idx)/deck.length)*100}%`, height:"100%", background:sub.color, borderRadius:50, transition:"width 0.3s" }}/>
              </div>
              <div style={{ background:T.card, border:`1.5px solid ${T.cardBorder}`, borderRadius:22, padding:"36px 24px", minHeight:200, display:"flex", flexDirection:"column", alignItems:"center", justifyContent:"center", textAlign:"center", boxShadow:`0 4px 20px ${T.shadow}`, transform:anim==="right"?"translateX(70px) rotate(4deg)":anim==="left"?"translateX(-70px) rotate(-4deg)":"none", opacity:anim?0:1, transition:"transform 0.26s, opacity 0.26s", marginBottom:18 }}>
                {(card.mastered||0)>=3&&<div style={{ fontSize:16, marginBottom:8 }}>👑</div>}
                <div style={{ fontSize:10, color:sub.color, letterSpacing:3, textTransform:"uppercase", marginBottom:14 }}>{shown?"Answer":"Question"}</div>
                <div style={{ color:T.text, fontSize:18, lineHeight:1.6, fontWeight:shown?400:700 }}>{shown?card.back:card.front}</div>
                {card.vaultName && <div style={{ fontSize:10, color:T.faint, marginTop:12 }}>From: {card.vaultName}</div>}
                {!shown&&<button onClick={()=>setShown(true)} style={{ marginTop:22, background:"none", border:`1.5px solid ${T.border}`, borderRadius:50, padding:"9px 24px", color:T.muted, fontSize:12, cursor:"pointer", fontWeight:700, fontFamily:"inherit" }}>Reveal →</button>}
              </div>
              {shown&&<div style={{ display:"flex", gap:12 }}>
                <button onClick={()=>answer(false)} style={{ flex:1, background:T.redBg, border:`1.5px solid ${T.redBd}`, borderRadius:14, padding:"15px 0", color:T.red, fontSize:14, cursor:"pointer", fontWeight:800, fontFamily:"inherit" }}>↺ Again</button>
                <button onClick={()=>answer(true)} style={{ flex:1, background:T.greenBg, border:`1.5px solid ${T.greenBd}`, borderRadius:14, padding:"15px 0", color:T.green, fontSize:14, cursor:"pointer", fontWeight:800, fontFamily:"inherit" }}>✓ Got it!</button>
              </div>}
            </div>
          );
        })()}

        {phase==="done" && (() => {
          const got=log.filter(r=>r.got).length, pct=Math.round((got/log.length)*100);
          return (
            <div style={{ textAlign:"center" }}>
              <div style={{ fontSize:56, marginBottom:10 }}>{pct===100?"🏆":pct>=80?"🎯":pct>=60?"💪":"📚"}</div>
              <div style={{ fontSize:42, fontWeight:900, color:T.accent }}>{pct}%</div>
              <div style={{ color:T.muted, fontSize:14, marginTop:4, marginBottom:20 }}>{got}/{log.length} correct</div>
              <div style={{ background:T.accentBg, border:`1px solid ${T.accent}44`, borderRadius:14, padding:14, marginBottom:20, textAlign:"left" }}>
                <div style={{ fontSize:10, color:T.accent, letterSpacing:2, textTransform:"uppercase", marginBottom:6 }}>📅 Spaced Repetition Updated</div>
                <div style={{ fontSize:12, color:T.muted, lineHeight:1.7 }}>Cards you got right have been scheduled further into the future. Cards you missed will appear sooner. This is the Ebbinghaus forgetting curve in action.</div>
              </div>
              <div style={{ textAlign:"left", marginBottom:20 }}>
                {log.map((r,i)=><div key={i} style={{ display:"flex", alignItems:"center", gap:10, background:T.card, border:`1px solid ${r.got?T.greenBd:T.redBd}`, borderRadius:10, padding:"9px 13px", marginBottom:5 }}>
                  <span style={{ color:r.got?T.green:T.red, fontSize:13 }}>{r.got?"✓":"↺"}</span>
                  <span style={{ flex:1, color:T.text, fontSize:13 }}>{r.front}</span>
                  {r.nextReview&&r.vaultId&&<span style={{ fontSize:10, color:T.faint }}>+{SR_INTERVALS[Math.min((r.mastered||0),5)]}d</span>}
                </div>)}
              </div>
              <Btn label="Close" onClick={onClose} block/>
            </div>
          );
        })()}
      </div>
    </div>
  );
};

  );

  const nav = [

    {id:"profile",   label:"Profile",    Icon:NavProfile},
  ];

  return (
    <ThemeCtx.Provider value={theme}>
